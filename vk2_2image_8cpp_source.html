42 Image::Image(ImageInfo info):
43 image(VK_NULL_HANDLE),
44 format(info.format),
45 layout(ImageLayout::Undefined),
46 tiling(info.image_tiling),
47 residency(info.residency),
48 dimensions(info.dimensions),
49 device(info.device),
50 destroy_on_destructor(true),
51 vma_alloc(VmaAllocation{}),
52 vma_alloc_info()
53 {
54 TZ_PROFZONE("Vulkan Backend - Image Create", TZ_PROFCOL_RED);
55 VkImageCreateInfo create
56 {
57 .sType = VK_STRUCTURE_TYPE_IMAGE_CREATE_INFO,
58 .pNext = nullptr,
59 .flags = 0,
60 .imageType = VK_IMAGE_TYPE_2D,
61 .format = static_cast<VkFormat>(this->format),
62 .extent = {.width = info.dimensions[0], .height = info.dimensions[1], .depth = 1},
63 .mipLevels = info.mip_levels,
64 .arrayLayers = info.array_layers,
65 .samples = static_cast<VkSampleCountFlagBits>(info.sample_count),
66 .tiling = static_cast<VkImageTiling>(info.image_tiling),
67 .usage = static_cast<VkImageUsageFlags>(static_cast<ImageUsage>(info.usage)),
68 .sharingMode = VK_SHARING_MODE_EXCLUSIVE,
69 .queueFamilyIndexCount = 0,
70 .pQueueFamilyIndices = nullptr,
71 .initialLayout = static_cast<VkImageLayout>(this->layout)
72 };
73
74 VmaMemoryUsage vma_usage;
75 VmaAllocationCreateFlags vma_flags = 0;
76 switch(info.residency)
77 {
78 default:
79 tz_error("Unrecognised MemoryResidency. Please submit a bug report.");
80 [[fallthrough]];
81 case MemoryResidency::GPU:
82 vma_usage = VMA_MEMORY_USAGE_GPU_ONLY;
83 break;
84 case MemoryResidency::CPU:
85 vma_usage = VMA_MEMORY_USAGE_CPU_ONLY;
86 break;
87 case MemoryResidency::CPUPersistent:
88 vma_usage = VMA_MEMORY_USAGE_CPU_ONLY;
89 vma_flags = VMA_ALLOCATION_CREATE_MAPPED_BIT;
90 break;
91 }
92
93 VmaAllocationCreateInfo alloc
94 {
95 .flags = vma_flags,
96 .usage = vma_usage,
97 .requiredFlags = 0,
98 .preferredFlags = 0,
99 .memoryTypeBits = 0,
100 .pool = {},
101 .pUserData = nullptr,
102 .priority = 0
103 };
104
105 VkResult res = vmaCreateImage(this->get_device().vma_native(), &create, &alloc, &this->image, &this->vma_alloc.value(), &this->vma_alloc_info);
106 switch(res)
107 {
108 case VK_SUCCESS:
109
110 break;
111 case VK_ERROR_OUT_OF_HOST_MEMORY:
112 tz_error("Failed to create Image because we ran out of host memory (RAM). Please ensure that your system meets the minimum requirements.");
113 break;
114 case VK_ERROR_OUT_OF_DEVICE_MEMORY:
115 tz_error("Failed to create Image because we ran out of device memory (VRAM). Please ensure that your system meets the minimum requirements.");
116 break;
117 case VK_ERROR_FEATURE_NOT_PRESENT:
118 tz_error("Undocumented return value VK_ERROR_FEATURE_NOT_PRESENT from vmaCreateImage. TODO: Invoke vkGetPhysicalDeviceImageFormatProperties before trying to create the image to catch this early. The combination of format, tiling, residency etc are unsupported for this specific machine. Please submit a bug report.");
119 break;
120 default:
121 tz_error("Failed to create Image but cannot determine why. Please submit a bug report.");
122 break;
123 }
124 }

184 void* Image::map()
185 {
186 TZ_PROFZONE("Vulkan Backend - Image Map", TZ_PROFCOL_RED);
187 if(!this->vma_alloc.has_value() || this->residency == MemoryResidency::GPU)
188 {
189 return nullptr;
190 }
191 if(this->tiling != ImageTiling::Linear)
192 {
193 tz_warning_report("Mapping image memory, but its tiling is not linear. You cannot predict the internal data format so writes are likely to be invalid");
194 }
195 // If we've mapped earlier/are CPUPersistent, we already have a ptr we can use.
196 if(this->vma_alloc_info.pMappedData != nullptr)
197 {
198 return this->vma_alloc_info.pMappedData;
199 }
200 // Otherwise we'll actually perform the map
201 VkResult res = vmaMapMemory(this->get_device().vma_native(), this->vma_alloc.value(), &this->vma_alloc_info.pMappedData);
202 if(res == VK_SUCCESS)
203 {
204 return this->vma_alloc_info.pMappedData;
205 }
206 return nullptr;
207 }
