206 void common_renderer_dbgui(RendererType auto& renderer)
207 {
208 if(renderer.is_null())
209 {
210 ImGui::Text("Null Renderer");
211 return;
212 }
213 ImGui::PushID(&renderer);
214 ImGui::Text("Renderer Name:");
215 ImGui::SameLine();
216 ImGui::TextColored(ImVec4{1.0f, 0.6f, 0.6f, 1.0f}, "%s", renderer.debug_get_name().data());
217 if(renderer.resource_count() > 0)
218 {
219 if(ImGui::CollapsingHeader("Resources"))
220 {
221 unsigned int rcount = renderer.resource_count() - 1;
222 ImGui::Text("Resource Count: %u", renderer.resource_count());
223 static int res_id = 0;
224 res_id = std::clamp(res_id, 0, static_cast<int>(rcount));
225 ImGui::SliderInt("Resource ID:", &res_id, 0, rcount);
226
227 // Display information about current resource.
228 ImGui::Indent();
229 renderer.get_resource(static_cast<hdk::hanval>(res_id))->dbgui();
230 ImGui::Unindent();
231 }
232 }
233 else
234 {
235 ImGui::Text("Renderer has no resources.");
236 }
237 if(ImGui::CollapsingHeader("Renderer State"))
238 {
239 ImGui::TextColored(ImVec4{1.0f, 0.6f, 0.6f, 1.0f}, "Graphics");
240 ImGui::Indent();
241 // Graphics - Index Buffer
242 {
243 auto han = renderer.get_state().graphics.index_buffer;
244 auto* comp = renderer.get_component(han);
245 if(comp != nullptr && ImGui::CollapsingHeader("Index Buffer"))
246 {
247 ImGui::Text("[Resource %zu]", static_cast<std::size_t>(static_cast<hdk::hanval>(han)));
248 ImGui::Indent();
249 comp->get_resource()->dbgui();
250 ImGui::Unindent();
251 }
252 }
253 // Graphics - Draw Buffer
254 {
255 auto han = renderer.get_state().graphics.draw_buffer;
256 auto* comp = renderer.get_component(han);
257 if(comp != nullptr && ImGui::CollapsingHeader("Draw Indirect Buffer"))
258 {
259 ImGui::Text("[Resource %zu]", static_cast<std::size_t>(static_cast<hdk::hanval>(han)));
260 ImGui::Indent();
261 comp->get_resource()->dbgui();
262 ImGui::Unindent();
263 }
264 }
265 ImGui::Spacing();
266 // Graphics - Clear Colour
267 {
268 auto col = renderer.get_state().graphics.clear_colour;
269 if(ImGui::DragFloat4("Clear Colour", col.data().data(), 0.02f, 0.0f, 1.0f))
270 {
271 ImGui::Text("Direct edit is not yet supported.");
272 }
273 }
274 // Graphics - Tri Count
275 {
276 if(renderer.get_state().graphics.draw_buffer != hdk::nullhand)
277 {
278 ImGui::Text("Triangle Count: Indirect Buffer");
279 }
280 else
281 {
282 ImGui::Text("Triangle Count: %zu", renderer.get_state().graphics.tri_count);
283 }
284 }
285 ImGui::Unindent();
286 ImGui::TextColored(ImVec4{1.0f, 0.6f, 0.6f, 1.0f}, "Compute");
287 ImGui::Indent();
288 // Compute - Kernel
289 {
290 auto kern = static_cast<hdk::vec3i>(renderer.get_state().compute.kernel);
291 if(ImGui::DragInt3("Kernel", kern.data().data(), 0.25f, 0, 64))
292 {
293 ImGui::Text("Direct edit is not yet supported.");
294 }
295 }
296 ImGui::Unindent();
297 }
298 if(!renderer.get_options().empty() && ImGui::CollapsingHeader("Renderer Options"))
299 {
300 for(RendererOption option : renderer.get_options())
301 {
302 ImGui::Text("%s", detail::renderer_option_strings[static_cast<int>(option)]);
303 }
304 #if HDK_DEBUG
305 ImGui::PushTextWrapPos();
306 ImGui::TextDisabled("Note: In debug builds, extra options might be present that you did not ask for. These are added to allow the debug-ui to display ontop of your rendered output.");
307 ImGui::PopTextWrapPos();
308 #endif //TZ_DEBUG
309 }
310 if(ImGui::CollapsingHeader("On-Demand Edits"))
311 {
312 ImGui::Text("Wireframe Mode");
313 if(ImGui::Button("Enable"))
314 {
315 renderer.edit
316 (
317 tz::gl::RendererEditBuilder{}

319 .build()
320 );
321 }
322 ImGui::SameLine();

328 .render_state({.wireframe_mode = false})
329 .build()
330 );
331 }
332 ImGui::Spacing();
333 }
334 ImGui::Separator();
335 ImGui::PopID();
336 }
