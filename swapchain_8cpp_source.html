22 // Ensure the swapchain info specifies meets hardware requirements:

24 tz_assert(ldev.get_extensions().contains(DeviceExtension::Swapchain), "Attempted to create Swapchain, but owning LogicalDevice does not have DeviceExtension::Swapchain enabled!");

35 // The PhysicalDevice might have a number for us, but it may not. If not we just use the window dimensions.
36 VkExtent2D extent;
37 {
38 if(pdev_surface_cap.maybe_surface_dimensions.has_value())

42 extent.height = current_extent[1];
43 }

48 }

53 create.surface = surface.native();

55 create.imageFormat = static_cast<VkFormat>(this->info.image_format);
56 create.imageColorSpace = VK_COLOR_SPACE_SRGB_NONLINEAR_KHR;

60 create.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
61 create.queueFamilyIndexCount = 0;
62 create.pQueueFamilyIndices = nullptr;

64 create.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
65 create.presentMode = static_cast<VkPresentModeKHR>(this->info.present_mode);
66 create.clipped = VK_TRUE;

79 case VK_ERROR_OUT_OF_HOST_MEMORY:
80 tz_error("Ran out of host memory (RAM) while trying to create Swapchain. Ensure your device meets minimum requirements.");
81 break;
82 case VK_ERROR_OUT_OF_DEVICE_MEMORY:
83 tz_error("Ran out of device memory (VRAM) while trying to create Swapchain. Ensure your device meets minimum requirements");
84 break;
85 case VK_ERROR_DEVICE_LOST:
86 tz_error("Device lost whilst trying to create a Swapchain. Possible hardware fault. Please be aware: Device loss is serious and further attempts to run the engine may cause serious hazards, such as operating system crash. Submit a bug report but do not attempt to reproduce the issue.");

90 break;

92 tz_error("Tried to create a Swapchain. Requested window is already in-use. If you're somehow using more than 1 window while running this application, this is probably why - Try reproducing the issue in a single window. If you're not doing anything special, please submit a bug report");
93 break;
94 case VK_ERROR_INITIALIZATION_FAILED:
95 tz_error("Tried to create a Swapchain, but something went wrong for implementation-specified reasons. Unfortunately this error code is extremely cryptic. Please submit a bug report (although please don't get your hopes up)");
96 break;
97 default:
98 tz_error("Failed to create a Swapchain, and the error code is unrecognised. Either we are missing a return code case, or something has gone very, very wrong. Please submit a bug report.");
99 break;
100 }
101

103 this->dimensions[1] = static_cast<unsigned int>(extent.height);
104 this->initialise_images();

219 void Swapchain::initialise_images()
220 {
221 tz_assert(!this->is_null(), "Tried to initialise Swapchain images, but the Swapchain is null. Please submit a bug report.");
222 if(!this->swapchain_images.empty())
223 {
224 this->swapchain_images.clear();
225 this->swapchain_image_views.clear();
226 }
227
228 std::uint32_t true_image_count;
229 vkGetSwapchainImagesKHR(this->get_device().native(), this->swapchain, &true_image_count, nullptr);
230 this->swapchain_images.reserve(true_image_count);
231 for(std::uint32_t i = 0; std::cmp_less(i, true_image_count); i++)
232 {
233 SwapchainImageInfo sinfo;
234 sinfo.swapchain = this;
235 sinfo.image_index = i;
236
237 Image& img = this->swapchain_images.emplace_back(sinfo);
238
239 ImageViewInfo view_info;
240 view_info.image = &img;
241 view_info.aspect = ImageAspect::Colour;
242 this->swapchain_image_views.emplace_back(view_info);
243 }
244 }
