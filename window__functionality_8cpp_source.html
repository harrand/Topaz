118 void WindowFunctionality::handle_key_event(int key, int scancode, int action, [[maybe_unused]] int mods)
119 {
120 KeyPressType t;
121 switch(action)
122 {
123 case GLFW_PRESS:
124 [[fallthrough]];
125 case GLFW_REPEAT:
126 t = KeyPressType::Press;
127 break;
128 case GLFW_RELEASE:
129 t = KeyPressType::Release;
130 break;
131 default:
132 return;
133 break;
134 }
135 KeyInfo tz_key = get_key_from_glfw_code(key, scancode);
136 if(tz_key == peripherals::keyboard::key_null)
137 {
138 // GLFW key doesn't translate cleanly into a TZ key. We don't support such keys.
139 return;
140 }
141 this->kb_state.update
142 ({
143 .key = tz_key,
144 .type = t
145 });
146 }

161 break;
162 case GLFW_MOUSE_BUTTON_MIDDLE:
163 tz_button = MouseButton::Middle;
164 break;
165 case GLFW_MOUSE_BUTTON_4:
166 tz_button = MouseButton::Extra0;
167 break;
168 case GLFW_MOUSE_BUTTON_5:
169 tz_button = MouseButton::Extra1;
170 break;
171 case GLFW_MOUSE_BUTTON_6:
172 tz_button = MouseButton::Extra2;
173 break;
174 case GLFW_MOUSE_BUTTON_7:
175 tz_button = MouseButton::Extra3;
176 break;
177 case GLFW_MOUSE_BUTTON_8:
178 tz_button = MouseButton::Extra4;
179 break;
180 }
181 MouseButtonInfo button_info = peripherals::mouse::get_mouse_button(tz_button);
182 if(button_info == peripherals::mouse::mouse_button_null)
183 {
184 return;
185 }
186 double mxpos, mypos;
187 glfwGetCursorPos(this->wnd, &mxpos, &mypos);
188 this->mb_state.update(

190 .button = button_info,
191 .press_position = this->mp_state.get_mouse_position()
192 }, action == GLFW_PRESS);

212 tz::KeyInfo get_key_from_glfw_code(int glfw_key, int scancode)
213 {
214 tz::KeyCode kc;
215 switch(glfw_key)
216 {
217 case GLFW_KEY_SPACE:
218 kc = tz::KeyCode::Space;
219 break;
220 case GLFW_KEY_LEFT_CONTROL:
221 kc = tz::KeyCode::LeftControl;
222 break;
223 case GLFW_KEY_LEFT_SHIFT:
224 kc = tz::KeyCode::LeftShift;
225 break;
226 case GLFW_KEY_LEFT_ALT:
227 kc = tz::KeyCode::LeftAlt;
228 break;
229 case GLFW_KEY_RIGHT_ALT:
230 kc = tz::KeyCode::AltGr;
231 break;
232 case GLFW_KEY_RIGHT_CONTROL:
233 kc = tz::KeyCode::RightControl;
234 break;
235 case GLFW_KEY_RIGHT_SHIFT:
236 kc = tz::KeyCode::RightShift;
237 break;
238 case GLFW_KEY_ENTER:
239 kc = tz::KeyCode::Enter;
240 break;
241 case GLFW_KEY_BACKSPACE:
242 kc = tz::KeyCode::Backspace;
243 break;
244 case GLFW_KEY_UP:
245 kc = tz::KeyCode::ArrowUp;
246 break;
247 case GLFW_KEY_LEFT:
248 kc = tz::KeyCode::ArrowLeft;
249 break;
250 case GLFW_KEY_DOWN:
251 kc = tz::KeyCode::ArrowDown;
252 break;
253 case GLFW_KEY_RIGHT:
254 kc = tz::KeyCode::ArrowRight;
255 break;
256 case GLFW_KEY_CAPS_LOCK:
257 kc = tz::KeyCode::CapsLock;
258 break;
259 case GLFW_KEY_TAB:
260 kc = tz::KeyCode::Tab;
261 break;
262 case GLFW_KEY_ESCAPE:
263 kc = tz::KeyCode::Escape;
264 break;
265 default:
266 // Try and retrieve by key name. This should catch all printable keys.
267 const char* glfw_key_name = glfwGetKeyName(glfw_key, scancode);
268 if(glfw_key_name == nullptr || glfw_key_name[0] == '\0')
269 {
270 return peripherals::keyboard::key_null;
271 }
272 return peripherals::keyboard::get_key(glfw_key_name[0]);
273 break;
274 }
275 return peripherals::keyboard::get_key(kc);
276 }
