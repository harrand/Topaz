115 Mat4 perspective(float fov, float aspect_ratio, float near, float far)
116 {
117 TZ_PROFZONE("Matrix Perspective", TZ_PROFCOL_BLUE);
118 const float thf = std::tan(fov / 2.0f);
119 Mat4 m = Mat4::identity();
120 m(0, 0) = 1.0f / (aspect_ratio * thf);
121 m(1, 1) = 1.0f / thf;
122
123 m(2, 2) = (far + near) / (near - far);
124 m(3, 2) = -1.0f;
125
126 m(2, 3) = (2.0f * far * near) / (near - far);
127 m(3, 3) = 0.0f;
128 return m;
129 }

132 {
133 TZ_PROFZONE("Matrix Orthographic", TZ_PROFCOL_BLUE);
134 Mat4 m = Mat4::identity();
135 m(0, 0) = 2.0f / (right - left);
136 m(1, 1) = 2.0f / (top - bottom);
137 m(2, 2) = -2.0f / (far - near);
138
139 Mat4::Row& bottom_row = m[2];
140 bottom_row[0] = -(right + left) / (right - left);
141 bottom_row[1] = -(top + bottom) / (top - bottom);
142 bottom_row[2] = -(far + near) / (far - near);
143 return m;
