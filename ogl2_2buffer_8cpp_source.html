114 Buffer clone_resized(const Buffer& buf, std::size_t new_size)
115 {
116 TZ_PROFZONE("OpenGL Backend - Buffer Clone Resized", TZ_PROFCOL_RED);
117 TZ_PROFZONE_GPU("Buffer Clone Resized", TZ_PROFCOL_RED);
118 Buffer newbuf
119 {{
120 .target = buf.get_target(),
121 .residency = buf.get_residency(),
122 .size_bytes = new_size
123 }};
124 if(new_size <= buf.size())
125 {
126
127 // We're smaller, so we copy all that we can.
128 glCopyNamedBufferSubData(buf.native(), newbuf.native(), 0, 0, static_cast<GLsizeiptr>(new_size));
129 }
130 else
131 {
132 // We're bigger or same size, so just copy the entire source. If anything is left in the destination it will be undefined values.
133 copy(buf, newbuf);
134 }
135 return newbuf;
136 }
