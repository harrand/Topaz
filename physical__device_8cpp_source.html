10 PhysicalDeviceSurfaceCapabilityInfo::PhysicalDeviceSurfaceCapabilityInfo(VkSurfaceCapabilitiesKHR vk_capabilities):

12 max_image_count(vk_capabilities.maxImageCount),
13 current_transform(vk_capabilities.currentTransform),
14 maybe_surface_dimensions(std::nullopt)
15 {
16 // currentExtent is the current width and height of the surface, or the special value (0xFFFFFFFF, 0xFFFFFFFF) indicating that the surface size will be determined by the extent of a swapchain targeting the surface.
17 const VkExtent2D cur = vk_capabilities.currentExtent;
18 constexpr std::uint32_t special_value = 0xFFFFFFFF;
19 if(cur.width == special_value && cur.height == special_value)
20 {
21 // Leave maybe_surface_dimensions as nullopt as it's upto the swapchain now.
22 }
23 else
24 {
25 // Retrieve the SwapchainExtent corresponding to the width and height of the surface.
26 SwapchainExtent dims;
27 dims.current_extent = tz::Vector<std::uint32_t, 2>{cur.width, cur.height};
28 const VkExtent2D min = vk_capabilities.minImageExtent;
29 const VkExtent2D max = vk_capabilities.maxImageExtent;
30 dims.min_image_extent = tz::Vector<std::uint32_t, 2>{min.width, min.height};
31 dims.max_image_extent = tz::Vector<std::uint32_t, 2>{max.width, max.height};
32 this->maybe_surface_dimensions = dims;

38 DeviceFeatureField to_feature_field(DeviceFeatureInfo features)
39 {
40 DeviceFeatureField ret;
41
42 if(features.features.features.multiDrawIndirect)
43 {
44 ret |= DeviceFeature::MultiDrawIndirect;
45 }
46 if(features.descriptor_indexing_features.descriptorBindingStorageImageUpdateAfterBind
47 && features.descriptor_indexing_features.descriptorBindingStorageBufferUpdateAfterBind
48 && features.descriptor_indexing_features.descriptorBindingSampledImageUpdateAfterBind
49 && features.descriptor_indexing_features.descriptorBindingUpdateUnusedWhilePending
50 && features.descriptor_indexing_features.descriptorBindingPartiallyBound
51 && features.descriptor_indexing_features.descriptorBindingVariableDescriptorCount)
52 {
53 ret |= DeviceFeature::BindlessDescriptors;
54 }
55 if(features.features12.timelineSemaphore)
56 {
57 ret |= DeviceFeature::TimelineSemaphores;
58 }
59 if(features.features.features.logicOp)
60 {
61 ret |= DeviceFeature::ColourBlendLogicalOperations;
62 }
63
64 return ret;

123 DeviceExtensionList PhysicalDevice::get_supported_extensions() const
124 {
125 tz_assert(!this->is_null(), "This was PhysicalDevice::null()");
126 std::uint32_t supported_extension_count;
127 vkEnumerateDeviceExtensionProperties(this->dev, nullptr, &supported_extension_count, nullptr);
128 std::vector<VkExtensionProperties> props;
129 std::vector<util::VkExtension> extensions;
130
131 props.resize(static_cast<decltype(props)::size_type>(supported_extension_count));
132 vkEnumerateDeviceExtensionProperties(this->dev, nullptr, &supported_extension_count, props.data());
133 extensions.resize(supported_extension_count);
134 std::transform(props.begin(), props.end(), extensions.begin(), [](const VkExtensionProperties& prop)->util::VkExtension{return prop.extensionName;});
135 DeviceExtensionList exts;
136 std::for_each(extensions.begin(), extensions.end(), [&exts](util::VkExtension ext)
137 {
138 DeviceExtension tz_ext = util::to_device_extension(ext);
139 if(tz_ext != DeviceExtension::Count)
140 {
141 exts |= tz_ext;
142 }
143 });
144
145 return exts;
146 }
