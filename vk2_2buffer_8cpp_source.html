122 void* Buffer::map()
123 {
124 TZ_PROFZONE("Vulkan Backend - Buffer Map", TZ_PROFCOL_RED);
125 // If we're GPU resident we can't be host visible, so early out.
126 if(this->info.residency == MemoryResidency::GPU)
127 {
128 return nullptr;
129 }
130 // If we've mapped earlier/are CPUPersistent, we already have a ptr we can use.
131 if(this->vma_alloc_info.pMappedData != nullptr)
132 {
133 return this->vma_alloc_info.pMappedData;
134 }
135 // Otherwise we'll actually perform the map.
136 VkResult res = vmaMapMemory(this->get_device().vma_native(), this->vma_alloc, &this->vma_alloc_info.pMappedData);
137 if(res == VK_SUCCESS)
138 {
139 return this->vma_alloc_info.pMappedData;
140 }
141 return nullptr;
142 }
