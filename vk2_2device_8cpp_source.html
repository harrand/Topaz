41 }
42 std::uint32_t swapchain_img_min = device.get_hardware().get_surface_capabilities().min_image_count;
43 std::uint32_t swapchain_img_max = device.get_hardware().get_surface_capabilities().max_image_count;
44 std::uint32_t swapchain_img_count = std::clamp<std::uint32_t>(3u, swapchain_img_min, swapchain_img_max);
45 swapchain_img_count = std::min<std::uint32_t>(swapchain_img_count, 3u);
46 this->window_buf = vk2::Swapchain
47 {{
48 .device = &device,
49 .swapchain_image_count_minimum = swapchain_img_count,

163 void DeviceWindowVulkan::on_resize(tz::Vec2ui dims)
164 {
165 TZ_PROFZONE("DeviceWindowVulkan OnResize", TZ_PROFCOL_YELLOW);
166 if(dims[0] == 0 || dims[0] == 0)
167 {
168 return;
169 }
170 // Assume we have a head, headless no support for resizeable output yet.
171 tz_assert(this->as_swapchain() != nullptr, "Resizeable output for headless applications is not yet supported.");
172 vk2::Swapchain& old_swapchain = *this->as_swapchain();
173 if(old_swapchain.get_dimensions() == dims)
174 {
175 // Dimensions didn't actually change. Early out and don't bother telling the renderers.
176 return;
177 }
178 vk2::Swapchain new_swapchain
179 {{
180 .device = &old_swapchain.get_device(),
181 .swapchain_image_count_minimum = static_cast<std::uint32_t>(old_swapchain.get_image_views().size()),
182 .image_format = old_swapchain.get_image_format(),
183 .present_mode = old_swapchain.get_present_mode(),
184 .old_swapchain = &old_swapchain
185 }};
186 std::swap(old_swapchain, new_swapchain);
187 // Now notify all renderers.
188 this->renderer_resize_callbacks
189 ({
190 .new_dimensions = dims,
191 .new_output_images = this->get_output_images()
192 });
193 }

228 DeviceVulkan::DeviceVulkan(const vk2::VulkanInstance& instance):
229 device(vk2::LogicalDevice::null()),
230 window_storage()
231 {
232 TZ_PROFZONE("Vulkan Frontend - DeviceVulkan Create", TZ_PROFCOL_YELLOW);
233 // First, create a LogicalDevice.
234 // TODO: Don't just choose a device at random.
235 vk2::PhysicalDevice pdev = vk2::get_all_devices(instance).front();
236
237 vk2::DeviceExtensionList dev_exts = {};
238 const vk2::DeviceFeatureField dev_feats =
239 {
240 vk2::DeviceFeature::BindlessDescriptors,
241 vk2::DeviceFeature::MultiDrawIndirect,
242 vk2::DeviceFeature::ColourBlendLogicalOperations
243 };
244 tz_assert(pdev.get_supported_features().contains(dev_feats), "One or both of DeviceFeatures 'BindlessDescriptors' and 'MultiDrawIndirect' are not supported by this machine/driver. Please ensure your machine meets the system requirements.");
245 if(!instance.is_headless())
246 {
247 dev_exts = {vk2::DeviceExtension::Swapchain};
248 }
249 #if TZ_DEBUG
250 dev_exts |= vk2::DeviceExtension::ShaderDebugPrint;
251 #endif
252 tz_assert(pdev.get_supported_extensions().contains(dev_exts), "One or more of the %zu required DeviceExtensions are not supported by this machine/driver. Please ensure your machine meets the system requirements.", dev_exts.count());
253 this->device =
254 {{
255 .physical_device = pdev,
256 .extensions = dev_exts,
257 .features = dev_feats
258 }};
259
260 // Now finally create the DeviceWindow.
261 this->window_storage = {this->device};
262 }
