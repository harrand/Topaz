43 ShaderModule::CompileResult ShaderModule::compile()
44 {
45 TZ_PROFZONE("OpenGL Backend - Shader Source Compile", TZ_PROFCOL_RED);
46 TZ_PROFZONE_GPU("ShaderModule Compile", TZ_PROFCOL_RED);
47 glCompileShader(this->shader);
48 GLint success;
49 glGetShaderiv(this->shader, GL_COMPILE_STATUS, &success);
50 if(success == GL_TRUE)

69 Shader::Shader(ShaderInfo info):
70 program(0),
71 modules(),
72 info(info)
73 {
74 TZ_PROFZONE("OpenGL Backend - Shader Create", TZ_PROFCOL_RED);
75 TZ_PROFZONE_GPU("Shader Create", TZ_PROFCOL_RED);
76 tz_assert(ogl2::is_initialised(), "Tried to create shader program because ogl2 was not initialised. Please submit a bug report.");
77 this->program = glCreateProgram();
78
79 this->modules.reserve(info.modules.length());
80 std::transform(info.modules.begin(), info.modules.end(), std::back_inserter(this->modules),
81 [](const ShaderModuleInfo& minfo)->ShaderModule
82 {
83 return {minfo};
84 });
85 for(ShaderModule& shader_module : this->modules)
86 {
87 glAttachShader(this->program, shader_module.native());
88 ShaderModule::CompileResult cpl = shader_module.compile();
89 tz_assert(cpl.success, "Shader Module Compilation failed: %s", cpl.info_log.c_str());
90 }
91 LinkResult lnk = this->link();
92 tz_assert(lnk.success, "Shader Program Linkage+Validation failed: %s", lnk.info_log.c_str());
93 }
