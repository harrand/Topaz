10 ResourceStorage::ResourceStorage(std::span<const IResource* const> resources):
11 AssetStorageCommon<IResource>(resources),
12 components(),
13 image_handles(),
14 bindless_image_storage_buffer(ogl2::Buffer::null())
15 {
16 TZ_PROFZONE("OpenGL Frontend - RendererOGL ResourceStorage Create", TZ_PROFCOL_RED);
17 for(std::size_t i = 0; i < this->count(); i++)
18 {
19 IResource* res = this->get(static_cast<tz::HandleValue>(i));
20 switch(res->get_type())
21 {
22 case ResourceType::Buffer:
23 this->components.push_back(std::make_unique<BufferComponentOGL>(*res));
24 {
25 BufferComponentOGL& buf = *static_cast<BufferComponentOGL*>(this->components.back().get());
26 ogl2::Buffer& buffer = buf.ogl_get_buffer();
27 switch(buf.get_resource()->get_access())
28 {
29 case ResourceAccess::StaticFixed:
30 {
31 // Create staging buffer and do a copy.
32 ogl2::Buffer staging
33 {{
34 .target = ogl2::BufferTarget::ShaderStorage,
35 .residency = ogl2::BufferResidency::Dynamic,
36 .size_bytes = buffer.size()
37 }};
38 std::span<std::byte> staging_data = staging.map_as<std::byte>();
39 std::span<const std::byte> resource_data = res->data();
40 std::copy(resource_data.begin(), resource_data.end(), staging_data.begin());
41 ogl2::buffer::copy(staging, buffer);
42 }
43 break;
44 case ResourceAccess::DynamicFixed:
45 [[fallthrough]];
46 case ResourceAccess::DynamicVariable:
47 {
48
49 // Tell the resource to use the buffer's data. Also copy whatever we had before.
50 std::span<const std::byte> initial_data = res->data();
51 std::span<std::byte> buffer_byte_data = buffer.map_as<std::byte>();
52 std::copy(initial_data.begin(), initial_data.end(), buffer_byte_data.begin());
53 res->set_mapped_data(buffer_byte_data);
54 }
55 break;
56 default:
57 tz_error("ResourceAccess for this buffer is not yet implemented.");
58 break;
59 }
60 }
61 if(res->get_access() == ResourceAccess::DynamicFixed)
62 {
63 // Tell the resource to use the buffer's data. Also copy whatever we had before.
64 std::span<const std::byte> initial_data = res->data();
65 std::span<std::byte> buffer_byte_data = static_cast<BufferComponentOGL*>(this->components.back().get())->ogl_get_buffer().map_as<std::byte>();
66 std::copy(initial_data.begin(), initial_data.end(), buffer_byte_data.begin());
67 res->set_mapped_data(buffer_byte_data);
68 }
69 break;
70 case ResourceType::Image:
71 this->components.push_back(std::make_unique<ImageComponentOGL>(*res));
72 {
73 ImageComponentOGL& img = *static_cast<ImageComponentOGL*>(this->components.back().get());
74 ogl2::Image& image = img.ogl_get_image();
75 image.set_data(img.get_resource()->data());
76 image.make_bindless();
77 this->image_handles.push_back(img.ogl_get_image().get_bindless_handle());
78 }
79 break;
80 default:
81 tz_error("Unrecognised ResourceType. Please submit a bug report.");
82 break;
83 }
84 }
85 this->fill_bindless_image_buffer();
86 }

107 void ResourceStorage::fill_bindless_image_buffer()
108 {
109 TZ_PROFZONE("OpenGL Frontend - RendererOGL ResourceStorage (Fill bindless image buffer)", TZ_PROFCOL_RED);
110 if(this->image_handles.empty())
111 {
112 return;
113 }
114 const std::size_t buf_size = this->image_handles.size() * sizeof(ogl2::Image::BindlessTextureHandle);
115 this->bindless_image_storage_buffer =
116 {{
117 .target = ogl2::BufferTarget::ShaderStorage,
118 .residency = ogl2::BufferResidency::Static,
119 .size_bytes = buf_size
120 }};
121 ogl2::Buffer temp_copy_buffer =
122 {{
123 .target = ogl2::BufferTarget::ShaderStorage,
124 .residency = ogl2::BufferResidency::Dynamic,
125 .size_bytes = buf_size
126 }};
127 {
128 std::span<ogl2::Image::BindlessTextureHandle> handle_data = temp_copy_buffer.map_as<ogl2::Image::BindlessTextureHandle>();
129 std::copy(this->image_handles.begin(), this->image_handles.end(), handle_data.begin());
130 }
131 ogl2::buffer::copy(temp_copy_buffer, this->bindless_image_storage_buffer);
132 }

134 void ResourceStorage::bind_buffers()
135 {
136 if(this->resource_count_of(ResourceType::Buffer) == 0)

139 }
140 std::size_t i = 0;
141 for(std::size_t j = 0; j < this->components.size(); j++)
142 {
143 IComponent* comp = this->components[j].get();
144 if(comp->get_resource()->get_type() == ResourceType::Buffer)
145 {
146 static_cast<BufferComponentOGL*>(comp)->ogl_get_buffer().bind_to_resource_id(i++);
147 }
148 }
149 tz_assert(i == this->resource_count_of(ResourceType::Buffer), "ResourceStorage::bind_buffers had some kind of logic error and didn't find all the buffers. Please submit a bug report.");
150 }

187 ogl2::Shader ShaderManager::make_shader(const ShaderInfo& sinfo) const
188 {
189 tz::BasicList<ogl2::ShaderModuleInfo> modules;
190 if(sinfo.has_shader(ShaderStage::Compute))
191 {
192 // Compute, we only care about the compute shader.
193 modules =
194 {
195 {
196 .type = ogl2::ShaderType::Compute,
197 .code = ogl2::OGLString(sinfo.get_shader(ShaderStage::Compute))
198 }
199 };
200 }
201 else
202 {
203 // Graphics, must contain a Vertex and Fragment shader.
204 tz_assert(sinfo.has_shader(ShaderStage::Vertex), "ShaderInfo must contain a non-empty vertex shader if no compute shader is present.");
205 tz_assert(sinfo.has_shader(ShaderStage::Fragment), "ShaderInfo must contain a non-empty fragment shader if no compute shader is present.");
206 modules =
207 {
208 {
209 .type = ogl2::ShaderType::Vertex,
210 .code = ogl2::OGLString(sinfo.get_shader(ShaderStage::Vertex))
211 },
212 {
213 .type = ogl2::ShaderType::Fragment,
214 .code = ogl2::OGLString(sinfo.get_shader(ShaderStage::Fragment))
215 }
216 };
217 }
218 return
219 {{
220 .modules = modules
221 }};
222 }

297 void RendererOGL::render()
298 {
299 TZ_PROFZONE("OpenGL Frontend - RendererOGL Render", TZ_PROFCOL_RED);
300 TZ_PROFZONE_GPU("RendererOGL Render", TZ_PROFCOL_RED);
301 this->resources.write_dynamic_images();
302 glClearColor(this->clear_colour[0], this->clear_colour[1], this->clear_colour[2], this->clear_colour[3]);
303 if(this->options.contains(RendererOption::NoDepthTesting))
304 {
305 glDisable(GL_DEPTH_TEST);
306 }
307 else
308 {
309 glEnable(GL_DEPTH_TEST);
310 glDepthFunc(GL_LESS);
311 }
312 if(this->options.contains(RendererOption::AlphaBlending))
313 {
314 glEnable(GL_BLEND);
315 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
316 }
317 else
318 {
319 glDisable(GL_BLEND);
320 }
321 this->output.set_render_target();
322
323 this->shader.use();
324 this->resources.bind_buffers();
325 if(this->resources.resource_count_of(ResourceType::Image) > 0)
326 {
327 this->resources.bind_image_buffer();
328 }
329 this->vao.draw(this->tri_count);

338 void RendererOGL::edit(const RendererEditRequest& edit_request)
339 {
340 TZ_PROFZONE("OpenGL Backend - RendererOGL Edit", TZ_PROFCOL_RED);
341 if(edit_request.component_edits.empty())
342 {
343 return;
344 }
345 for(const RendererComponentEditRequest& component_edit : edit_request.component_edits)
346 {
347
348 std::visit(
349 [this](auto&& arg)
350 {
351 using T = std::decay_t<decltype(arg)>;
352 if constexpr(std::is_same_v<T, RendererBufferComponentEditRequest>)
353 {
354 auto buf_comp = static_cast<BufferComponentOGL*>(this->get_component(arg.buffer_handle));
355 tz_assert(buf_comp->get_resource()->get_access() == ResourceAccess::DynamicVariable, "Detected attempted resize of buffer resource (id %zu), but it ResourceAccess is not DynamicVariable. This means it is a fixed-size resource, so attempting to resize it is invalid.", static_cast<std::size_t>(static_cast<tz::HandleValue>(arg.buffer_handle)));
356 // Make new buffer copy, and swap them with the component's held buffer. That is literally it I believe.
357 ogl2::Buffer& old_buffer = buf_comp->ogl_get_buffer();
358 old_buffer = ogl2::buffer::clone_resized(old_buffer, arg.size);
359 // If we were dynamic, the resource mapping needs to refer to the new buffer though.
360 if(buf_comp->get_resource()->get_access() == ResourceAccess::DynamicVariable)
361 {
362 buf_comp->get_resource()->set_mapped_data(old_buffer.map_as<std::byte>());
363 }
364 }
365 else if constexpr(std::is_same_v<T, RendererImageComponentEditRequest>)
366 {
367 tz_error("Resizing images is not yet implemented");
368 }
369 else
370 {
371 tz_error("Unsupported variant type");
372 }
373 }, component_edit);
374 }
375 }
