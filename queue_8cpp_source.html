35 void Queue::submit(SubmitInfo submit_info)
36 {
37 TZ_PROFZONE("Vulkan Backend - Queue Submit", TZ_PROFCOL_RED);
38 if(submit_info.execution_complete_fence != nullptr)
39 {
40 tz_assert(!submit_info.execution_complete_fence->is_signalled(), "SubmitInfo contained Fence to signal on execution complete, but it was already signalled.");
41 }
42
43 std::vector<BinarySemaphore::NativeType> wait_sem_natives(submit_info.waits.length());
44 std::vector<VkPipelineStageFlags> wait_stage_natives(submit_info.waits.length());
45 for(std::size_t i = 0; i < submit_info.waits.length(); i++)
46 {
47 wait_sem_natives[i] = submit_info.waits[i].wait_semaphore->native();
48 wait_stage_natives[i] = static_cast<VkPipelineStageFlags>(submit_info.waits[i].wait_stage);
49 }
50
51 std::vector<CommandBuffer::NativeType> command_buffer_natives(submit_info.command_buffers.length());
52 std::transform(submit_info.command_buffers.begin(), submit_info.command_buffers.end(), command_buffer_natives.begin(), [](const CommandBuffer* buf){return buf->native();});
53
54 std::vector<BinarySemaphore::NativeType> signal_sem_natives(submit_info.signal_semaphores.length());
55 std::transform(submit_info.signal_semaphores.begin(), submit_info.signal_semaphores.end(), signal_sem_natives.begin(), [](const BinarySemaphore* signal_sem){return signal_sem->native();});
56
57 VkSubmitInfo sub
58 {
59 .sType = VK_STRUCTURE_TYPE_SUBMIT_INFO,
60 .pNext = nullptr,
61 .waitSemaphoreCount = static_cast<std::uint32_t>(submit_info.waits.length()),
62 .pWaitSemaphores = wait_sem_natives.data(),
63 .pWaitDstStageMask = wait_stage_natives.data(),
64 .commandBufferCount = static_cast<std::uint32_t>(submit_info.command_buffers.length()),
65 .pCommandBuffers = command_buffer_natives.data(),
66 .signalSemaphoreCount = static_cast<std::uint32_t>(submit_info.signal_semaphores.length()),
67 .pSignalSemaphores = signal_sem_natives.data()
68 };
69
70 VkFence fence_native = VK_NULL_HANDLE;
71 if(submit_info.execution_complete_fence != nullptr)
72 {
73 fence_native = submit_info.execution_complete_fence->native();
74 }
75 VkResult res = vkQueueSubmit(this->queue, 1, &sub, fence_native);
76 switch(res)
77 {
78 case VK_SUCCESS:
79 // Some commands require cpu-side operations too.
80 for(const CommandBuffer* buf : submit_info.command_buffers)
81 {
82 this->execute_cpu_side_command_buffer(*buf);
83 }
84 break;
85 case VK_ERROR_OUT_OF_HOST_MEMORY:
86 tz_error("Failed to submit Queue because we ran out of host memory (RAM). Please ensure that your system meets the minimum requirements.");
87 break;
88 case VK_ERROR_OUT_OF_DEVICE_MEMORY:
89 tz_error("Failed to submit Queue because we ran out of device memory (VRAM). Please ensure that your system meets the minimum requirements.");
90 break;
91 case VK_ERROR_DEVICE_LOST:
92 tz_error("Failed to submit Queue because device was lost. This is a fatal error.");
93 break;
94 default:
95 tz_error("Failed to submit Queue but cannot determine why. Please submit a bug report.");
96 break;
97 }

165 void Queue::execute_cpu_side_command_buffer(const CommandBuffer& command_buffer) const
166 {
167 TZ_PROFZONE("Vulkan Backend - Command Buffer CPU Execute", TZ_PROFCOL_RED);
168 for(const VulkanCommand::Variant& cmd : command_buffer.get_recorded_commands())
169 {
170 std::visit([](auto&& val)
171 {
172 using T = std::decay_t<decltype(val)>;
173 if constexpr(std::is_same_v<T, VulkanCommand::BeginRenderPass>)
174 {
175 TZ_PROFZONE("Vulkan Backend - RenderPass Begin CPU Execute", TZ_PROFCOL_RED);
176 // Change image's layout CPU-side.
177 Framebuffer& framebuffer = *val.framebuffer;
178 const RenderPass& pass = framebuffer.get_pass();
179 for(std::size_t i = 0; i < framebuffer.get_attachment_views().length(); i++)
180 {
181 Image& img = framebuffer.get_attachment_views()[i]->get_image();
182 ImageLayout output_layout = pass.get_info().attachments[i].initial_layout;
183 img.set_layout(output_layout);
184 }
185 }
186 if constexpr(std::is_same_v<T, VulkanCommand::EndRenderPass>)
187 {
188 TZ_PROFZONE("Vulkan Backend - RenderPass End CPU Execute", TZ_PROFCOL_RED);
189 // Change image's layout CPU-side.
190 Framebuffer& framebuffer = *val.framebuffer;
191 const RenderPass& pass = framebuffer.get_pass();
192 for(std::size_t i = 0; i < framebuffer.get_attachment_views().length(); i++)
193 {
194 Image& img = framebuffer.get_attachment_views()[i]->get_image();
195 ImageLayout output_layout = pass.get_info().attachments[i].final_layout;
196 img.set_layout(output_layout);
197 }
198 }
199 else if constexpr(std::is_same_v<T, VulkanCommand::TransitionImageLayout>)
200 {
201 TZ_PROFZONE("Vulkan Backend - TransitionImageLayout CPU Execute", TZ_PROFCOL_RED);
202 // We probably need to change the image's layout CPU-side.
203 Image& img = *val.image;
204 img.set_layout(val.target_layout);
205 }
206 }, cmd);
207 }
208 }
