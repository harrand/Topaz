31 bool RenderPassInfo::values_make_sense() const
32 {
33 std::size_t num_attachments = this->attachments.length();
34 auto stays_in_range = [num_attachments](const AttachmentReference& ref) -> bool
35 {
36 return ref.attachment_idx < num_attachments;
37 };
38
39 bool sane = true;
40 sane &= std::all_of(this->subpasses.begin(), this->subpasses.end(),
41 [stays_in_range](const RenderPassInfo::Subpass& subpass)->bool
42 {
43 bool depth_in_range_if_exist = true;
44 if(subpass.depth_stencil_attachment.has_value())
45 {
46 depth_in_range_if_exist = stays_in_range(subpass.depth_stencil_attachment.value());
47 }
48
49 auto input_stays_in_range = [stays_in_range](const RenderPassInfo::InputAttachmentReference& input_ref)->bool{return stays_in_range(input_ref.attachment_ref);};
50 return std::all_of(subpass.input_attachments.begin(), subpass.input_attachments.end(), input_stays_in_range)
51 && std::all_of(subpass.colour_attachments.begin(), subpass.colour_attachments.end(), stays_in_range) && depth_in_range_if_exist;
52 });
53
54 return sane;
55 }

57 RenderPass::RenderPass(RenderPassInfo info):
58 pass(VK_NULL_HANDLE),
59 info(info)
60 {
61 TZ_PROFZONE("Vulkan Backend - RenderPass Create", TZ_PROFCOL_RED);
62 tz_assert(this->info.has_valid_device(), "RenderPassInfo contained nullptr or null LogicalDevice. Please submit a bug report.");
63 tz_assert(this->info.values_make_sense(), "RenderPassInfo contained values which didn't pass sanity checks. This will most likely crash horribly. Please submit a bug report. Most likely candidate is one of the attachment references contains an attachment id out of range.");
64 std::vector<VkAttachmentDescription2> attachment_natives(info.attachments.length());
65 std::transform(info.attachments.begin(), info.attachments.end(), attachment_natives.begin(),
66 [](const Attachment& attachment) -> VkAttachmentDescription2
67 {
68 return
69 {
70 .sType = VK_STRUCTURE_TYPE_ATTACHMENT_DESCRIPTION_2,
71 .pNext = nullptr,
72 .flags = 0,
73 .format = static_cast<VkFormat>(attachment.format),
74 .samples = static_cast<VkSampleCountFlagBits>(attachment.sample_count),
75 .loadOp = static_cast<VkAttachmentLoadOp>(attachment.colour_depth_load),
76 .storeOp = static_cast<VkAttachmentStoreOp>(attachment.colour_depth_store),
77 .stencilLoadOp = static_cast<VkAttachmentLoadOp>(attachment.stencil_load),
78 .stencilStoreOp = static_cast<VkAttachmentStoreOp>(attachment.stencil_store),
79 .initialLayout = static_cast<VkImageLayout>(attachment.initial_layout),
80 .finalLayout = static_cast<VkImageLayout>(attachment.final_layout)
81 };
82 });
83
84 // The VKAPI stuff gets really painful here, lots of pointers within pointers. We want one big vector for all, so we find the upper bound and reserve here.
85 std::vector<VkAttachmentReference2> input_attachment_natives(info.total_input_attachment_count());
86 std::vector<VkAttachmentReference2> colour_attachment_natives(info.total_colour_attachment_count());
87 std::vector<std::optional<VkAttachmentReference2>> depth_attachment_natives(info.subpasses.length(), std::nullopt);
88
89 std::vector<VkSubpassDescription2> subpass_natives(info.subpasses.length());
90 struct AttachmentOffset
91 {
92 std::size_t input = 0;
93 std::size_t colour = 0;
94 std::size_t depth_stencil = 0;
95 };
96 AttachmentOffset offset;
97 std::transform(info.subpasses.begin(), info.subpasses.end(), subpass_natives.begin(),
98 [&offset, &input_attachment_natives, &colour_attachment_natives, &depth_attachment_natives](const RenderPassInfo::Subpass& subpass) -> VkSubpassDescription2
99 {
100 // Fill in the attachment natives.
101 auto input_begin = input_attachment_natives.begin() + offset.input;
102 std::transform(subpass.input_attachments.begin(), subpass.input_attachments.end(), input_begin,
103 [](const RenderPassInfo::InputAttachmentReference& ref)->VkAttachmentReference2
104 {
105 return
106 {
107 .sType = VK_STRUCTURE_TYPE_ATTACHMENT_REFERENCE_2,
108 .pNext = nullptr,
109 .attachment = ref.attachment_ref.attachment_idx,
110 .layout = static_cast<VkImageLayout>(ref.attachment_ref.current_layout),
111 .aspectMask = static_cast<VkImageAspectFlags>(static_cast<ImageAspectFlag>(ref.aspect_flags))
112 };
113 });
114
115 // Now the colour natives.
116 auto colour_begin = colour_attachment_natives.begin() + offset.colour;
117 std::transform(subpass.colour_attachments.begin(), subpass.colour_attachments.end(), colour_begin,
118 [](const RenderPassInfo::AttachmentReference& ref)->VkAttachmentReference2
119 {
120 return
121 {
122 .sType = VK_STRUCTURE_TYPE_ATTACHMENT_REFERENCE_2,
123 .pNext = nullptr,
124 .attachment = ref.attachment_idx,
125 .layout = static_cast<VkImageLayout>(ref.current_layout),
126 .aspectMask = 0
127 };
128 });
129
130 const std::optional<RenderPassInfo::AttachmentReference>& maybe_depth_attachment_ref = subpass.depth_stencil_attachment;
131
132 // Then, the depth natives.
133 if(maybe_depth_attachment_ref.has_value())
134 {
135 std::uint32_t att_idx = VK_ATTACHMENT_UNUSED;
136 VkImageLayout depth_stencil_layout = static_cast<VkImageLayout>(ImageLayout::Undefined);
137 if(maybe_depth_attachment_ref.has_value())
138 {
139 att_idx = maybe_depth_attachment_ref.value().attachment_idx;
140 depth_stencil_layout = static_cast<VkImageLayout>(maybe_depth_attachment_ref.value().current_layout);
141 }
142 depth_attachment_natives[offset.depth_stencil] =
143 {
144 .sType = VK_STRUCTURE_TYPE_ATTACHMENT_REFERENCE_2,
145 .pNext = nullptr,
146 .attachment = att_idx,
147 .layout = depth_stencil_layout,
148 .aspectMask = 0
149 };
150 }
151
152 // Finally, write to the create natives.
153 const std::optional<VkAttachmentReference2>& maybe_depth = depth_attachment_natives[offset.depth_stencil];
154 VkSubpassDescription2 desc
155 {
156 .sType = VK_STRUCTURE_TYPE_SUBPASS_DESCRIPTION_2,
157 .pNext = nullptr,
158 .flags = 0,
159 .pipelineBindPoint = static_cast<VkPipelineBindPoint>(subpass.context),
160 .viewMask = 0,
161 .inputAttachmentCount = static_cast<std::uint32_t>(subpass.input_attachments.length()),
162 .pInputAttachments = input_attachment_natives.data() + offset.input,
163 .colorAttachmentCount = static_cast<std::uint32_t>(subpass.colour_attachments.length()),
164 .pColorAttachments = colour_attachment_natives.data() + offset.colour,
165 .pResolveAttachments = nullptr,
166 .pDepthStencilAttachment = maybe_depth.has_value() ? &maybe_depth.value() : nullptr,
167 .preserveAttachmentCount = 0,
168 .pPreserveAttachments = nullptr
169 };
170 offset.input += subpass.input_attachments.length();
171 offset.colour += subpass.colour_attachments.length();
172 offset.depth_stencil += subpass.depth_stencil_attachment.has_value() ? 1 : 0;
173 return desc;
174 });
175
176 VkSubpassDependency2 dep
177 {
178 .sType = VK_STRUCTURE_TYPE_SUBPASS_DEPENDENCY_2,
179 .pNext = nullptr,
180 .srcSubpass = VK_SUBPASS_EXTERNAL,
181 .dstSubpass = 0,
182 .srcStageMask = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT,
183 .dstStageMask = VK_PIPELINE_STAGE_COLOR_ATTACHMENT_OUTPUT_BIT,
184 .srcAccessMask = 0,

186 .dependencyFlags = 0,
187 .viewOffset = 0
188 };
