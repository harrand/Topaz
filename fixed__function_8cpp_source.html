20 ViewportState create_basic_viewport(tz::Vec2 dimensions)
21 {
22 ViewportState state;
23 state.viewports.add({});
24 VkViewport& viewport = state.viewports.back();
25
26 // OGL-like coordinate system. Note that ogl [mindepth, maxdepth] is [-1, 1] but we are forced to have it [0, 1] without extensions. This means we have no safe way of matching behaviour. For now the shaders have to do this manually :(
27 viewport.x = 0.0f;
28 viewport.y = dimensions[1];
29 viewport.width = dimensions[0];
30 viewport.height = -dimensions[1];
31 viewport.minDepth = 0.0f;
32 viewport.maxDepth = 1.0f;
33
34 state.scissors.add({});
35 VkRect2D& scissor = state.scissors.back();
36 scissor.offset = {0, 0};
37 scissor.extent = {static_cast<std::uint32_t>(dimensions[0]), static_cast<std::uint32_t>(dimensions[1])};
38
39 return state;
40 }
